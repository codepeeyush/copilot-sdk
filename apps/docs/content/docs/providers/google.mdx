## Setup

### 1. Get API Key

Get your API key from [Google AI Studio](https://aistudio.google.com/apikey)

### 2. Add Environment Variable

```bash title=".env.local"
GOOGLE_API_KEY=...
```

### 3. Create API Route (Backend)

```ts title="app/api/chat/route.ts"
import { createRuntime, createGoogle } from '@yourgpt/llm-sdk';

const google = createGoogle({
  apiKey: process.env.GOOGLE_API_KEY!,
});

const runtime = createRuntime({
  provider: google,
  model: 'gemini-1.5-pro',
  systemPrompt: 'You are a helpful assistant.',
});

export async function POST(request: Request) {
  return runtime.handleRequest(request);
}
```

### 4. Add Provider (Frontend)

```tsx title="app/providers.tsx"
'use client';

import { CopilotProvider } from '@yourgpt/copilot-sdk/react';

export function Providers({ children }: { children: React.ReactNode }) {
  return (
    <CopilotProvider runtimeUrl="/api/chat">
      {children}
    </CopilotProvider>
  );
}
```

## Configuration Options

```ts title="app/api/chat/route.ts"
const runtime = createRuntime({
  provider: google,
  model: 'gemini-1.5-pro',
  systemPrompt: 'You are a helpful assistant.',
  temperature: 0.7,        // 0-1
  maxTokens: 8192,         // Max response length
});
```

## Massive Context Window

Gemini supports 1M+ token context:

```ts title="app/api/chat/route.ts"
const runtime = createRuntime({
  provider: google,
  model: 'gemini-1.5-pro',
  systemPrompt: `Here is the entire codebase:

  ${entireCodebase}  // Can be 500K+ tokens!

  Help users understand and modify this code.`,
});
```

## Tool Calling

```tsx
import { useToolWithSchema } from '@yourgpt/copilot-sdk/react';
import { z } from 'zod';

function SearchTool() {
  useToolWithSchema({
    name: 'search_knowledge',
    description: 'Search internal knowledge base',
    schema: z.object({
      query: z.string(),
      filters: z.object({
        category: z.string().optional(),
        dateRange: z.string().optional(),
      }).optional(),
    }),
    handler: async ({ query, filters }) => {
      const results = await searchKnowledge(query, filters);
      return { success: true, data: results };
    },
  });

  return null;
}
```
