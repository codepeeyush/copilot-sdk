## Setup

### 1. Get API Key

Get your API key from [platform.openai.com](https://platform.openai.com/api-keys)

### 2. Add Environment Variable

```bash title=".env.local"
OPENAI_API_KEY=sk-...
```

### 3. Create API Route (Backend)

```ts title="app/api/chat/route.ts"
import { createRuntime, createOpenAI } from '@yourgpt/llm-sdk';

const openai = createOpenAI({
  apiKey: process.env.OPENAI_API_KEY!,
});

const runtime = createRuntime({
  provider: openai,
  model: 'gpt-4o-mini',
  systemPrompt: 'You are a helpful assistant.',
});

export async function POST(request: Request) {
  return runtime.handleRequest(request);
}
```

### 4. Add Provider (Frontend)

```tsx title="app/providers.tsx"
'use client';

import { CopilotProvider } from '@yourgpt/copilot-sdk/react';

export function Providers({ children }: { children: React.ReactNode }) {
  return (
    <CopilotProvider runtimeUrl="/api/chat">
      {children}
    </CopilotProvider>
  );
}
```

## Configuration Options

```ts title="app/api/chat/route.ts"
const runtime = createRuntime({
  provider: openai,
  model: 'gpt-4o',
  systemPrompt: 'You are a helpful assistant.',
  temperature: 0.7,        // 0-2, default 1
  maxTokens: 4096,         // Max response length
});
```

## Vision (Image Input)

GPT-4o supports image analysis:

```tsx
import { useCopilot } from '@yourgpt/copilot-sdk/react';

const { sendMessage } = useCopilot();

// Send message with image
sendMessage("What's in this image?", {
  attachments: [
    {
      type: 'image',
      data: base64ImageData,
      mimeType: 'image/png',
    }
  ]
});
```

## Tool Calling

OpenAI has excellent tool/function calling support:

```tsx
import { useToolWithSchema } from '@yourgpt/copilot-sdk/react';
import { z } from 'zod';

function SearchTool() {
  useToolWithSchema({
    name: 'search_products',
    description: 'Search for products',
    schema: z.object({
      query: z.string(),
      category: z.string().optional(),
    }),
    handler: async ({ query, category }) => {
      const results = await searchProducts(query, category);
      return { success: true, data: results };
    },
  });

  return null;
}
```
